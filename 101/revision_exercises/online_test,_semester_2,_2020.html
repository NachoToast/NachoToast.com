    def is_between_and_even(test_value, low_value, high_value):
        if test_value < low_value:
            return False
        if test_value > high_value:
            return False
        if test_value % 2 != 0:
            return False
        return True

    def get_zero_ending_numbers_count(string_of_numbers):
        count = 0
        while string_of_numbers.find(" ") != -1:
            next_space = string_of_numbers.find(" ")
            check_num = string_of_numbers[:next_space]
            if check_num[-1] == "0":
                count += 1
            string_of_numbers= string_of_numbers[next_space + 1:]
        return count

    def check_palindrome(text):
        i = 0
        while i < len(text):
            if text[i].isalpha():
                text = text[:i] + text[i].upper() + text[i + 1:]
                i += 1
            else:
                text = text[:i] + text[i + 1:]
        i = len(text) - 1
        reversed = ""
        while i >= 0:
            reversed += text[i]
            i -= 1
        if text == reversed:
            return True
        return False

    def word_puzzle(horizontal_str, vertical_str):
        i = 0
        x = -1
        y = -1
        while i < len(horizontal_str):
            index = vertical_str.find(horizontal_str[i])
            if index != -1:
                x = i
                y = index
                i = len(horizontal_str)
            i += 1
        if x == -1:
            return print("The two strings do not intersect")
        i = 0
        while i < len(vertical_str):
            if i == y:
                print(horizontal_str)
            else:
                print(" " * x + vertical_str[i])
            i += 1
