    def find_item_index(items_list, code):
        for i in range(len(items_list)):
            if items_list[i].find(code) != -1:
                return i
        print("The item", code, "does not exist.")
        return -1

    def get_total(cart_list):
        total = 0
        for i in range(len(cart_list)):
            sublist = cart_list[i].split(",")
            total += float(sublist[-2])
        return round(total, 2)

    def save_stock(items_list):
        output_file = open("ldou933.txt", "w")
        for item in items_list:
            output_file.write(item + "\n")
        output_file.close()

    def get_biggest_3(filename):
        input_file = open(filename, "r")
        contents = input_file.read().split()
        input_file.close()
        for i in range(len(contents)):
            contents[i] = int(contents[i])
        biggest1 = max(contents)
        contents.pop(contents.index(max(contents)))
        biggest2 = max(contents)
        contents.pop(contents.index(max(contents)))
        biggest3 = max(contents)
        contents.pop(contents.index(max(contents)))
        return [biggest3, biggest2, biggest1]

    def get_list_of_maximums(filename):
        input_file = open(filename, "r")
        contents = input_file.read().split("\n")
        input_file.close()
        output = []
        for num_list in contents:
            mini_list = num_list.split()
            for i in range(len(mini_list)):
                mini_list[i] = int(mini_list[i])
            output.append(max(mini_list))
        return output
