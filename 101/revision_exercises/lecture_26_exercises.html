    def print_mirrored_right_triangle(number_of_rows):
        for row in range(number_of_rows):
            for column in range(number_of_rows - row - 1):
                print(" ", end = "")
            print("*" * (row + 1), end = "")
            print()

    def print_special_rectangle(number_of_rows, number_of_columns):
        for row in range(number_of_rows):
            if row % 2 == 0:
                symbols = "-*"
            else:
                symbols = "*-"
            for column in range(number_of_columns):
                print(symbols[column % 2], end = "")
            print()

    def print_inverted_special_right_triangle(number_of_rows):

                symbols = "-*"
            for column in range(number_of_rows - row):

            print()

    def print_dot_columns (dot_list):
        max_in_list = max(dot_list)
        for current_number in range(max_in_list, 0, -1):
            for index in range(len(dot_list)):
                if dot_list[index] >= current_number:
                    print(". ", end = "")
                else:
                    print("  ", end = "")
            print()
