    def print_dict(a_dict):
        keys = a_dict.keys()
        for i in sorted(keys):
            print(i, ":", a_dict[i])

    def print_dict_in_order(a_dict):
        keys = sorted(a_dict.keys())
        for i in keys:
            print(i, ":", sorted(a_dict[i]))

    def create_marks_dictionary(list_of_tuples):
        a_dict = {}
        for (q, m) in list_of_tuples:
            if q in a_dict:
                a_dict[q].append(m)
            else:
                a_dict[q] = [m]
        return a_dict

    def create_vowels_dictionary(words):
        vowels_dict = {"a":[], "e":[], "i":[], "o":[], "u":[]}
        for i in words.lower().split():
            if i[0] in vowels_dict:
                if i not in vowels_dict[i[0]]:
                    vowels_dict[i[0]].append(i)
        return vowels_dict

    def word_count(sentence):
        a_dict = {}
        for i in sorted(sentence.lower().split()):
            if i in a_dict:
                a_dict[i] += 1
            else:
                a_dict[i] = 1
        for key in a_dict:
            occ = a_dict[key]
            if occ == 1:
                pl = "time."
            else:
                pl = "times."
            print(key, "appears", occ, pl)
