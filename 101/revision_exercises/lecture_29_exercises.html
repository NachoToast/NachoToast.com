    def print_title(word):
        dashes = 0
        word = word.upper()
        while len(word) > 0:
            print("-" * dashes + word)
            dashes += 1
            word = word[1:-1]

    def convert_first_letter(names):
        for i in range(len(names)):
            names[i] = names[i][0].upper() + names[i][1:]
        return names

    def sum_over(a_list_of_lists, target):
        sum = 0
        for list in a_list_of_lists:
            for num in list:
                if num > target:
                    sum += num
        return sum

    def get_left_most_x(points):
        smallest_x = points[0][0]
        for (x, y) in points:
            if x < smallest_x:
                smallest_x = x
        return smallest_x

    def merge(dict1, dict2):
        for key in dict2:
            if key in dict1:
                for value in dict2[key]:
                    if value not in dict1[key]:
                        dict1[key].append(value)
                        dict1[key].sort()
        return dict1

    def get_lines_from_file(filename):
        input_file = open(filename, 'r')
        contents = input_file.read().split('\n')
        input_file.close()
        return contents

    def print_pyramid(number):
        max = 2 * number + 1
        print("+" * max)
        for row in range(number):
            spaces = 2 * row + 1
            surround = int((max - spaces) / 2)
            print(surround * "+" + spaces * " " + surround * "+")
        print("+" * max)
