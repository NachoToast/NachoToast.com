    def check_password(password):
        symbol_str = ".;!*?"
        if len(password) < 8:
            return False
        i = 0
        symbols = 0
        numbers = 0
        letters = 0
        while i < len(password):
            if password[i].isdigit():
                numbers += 1
            elif password[i].isalpha():
                letters += 1
            elif symbol_str.find(password[i]) != -1:
                symbols += 1
            else:
                return False
            i += 1
        if symbols == 0 or numbers == 0 or letters == 0:
            return False
        return True

    def update_guess(answer, guess, character):

    def process_code(code_str):
        output = 0
        i = 0
        while i < len(code_str):
            subsequent = int(code_str[i])
            output += int(code_str[i + 1: i + subsequent + 1])
            i += subsequent + 1
        return output
