    def sum_of_smallest_two(num1, num2, num3):
        small_1 = min(num1, num2, num3)
        small_2 = num1 + num2 + num3 - max(num1, num2, num3) - small_1
        return small_1 + small_2

    def get_number(prompt):
        return int(input(prompt + ": "))
    def get_cost(distance, days, cost_per_k, cost_per_day):
        discount = get_discount(distance, days)
        cost = distance * cost_per_k + days * cost_per_day - discount
        return round(cost)
    def get_discount(distance, days):
        if days > 10 or distance > 5000:
            return 100
        return 0
    def display_summary(num_k, num_days, total_cost):
        discount = get_discount(num_k, num_days)
        print("Distance:", num_k, "kilometres")
        print("Days:", num_days, "days")
        if discount > 0:
            print("Discount: $", discount, sep="")
        print("Total: $", total_cost, sep="")
    main()

    def get_middle(n1, n2, n3):
        low = min(n1, n2, n3)
        high = max(n1, n2, n3)
        return n1 + n2 + n3 - low - high

    def get_digit_sum(number_str):
        return int(number_str[0]) + int(number_str[1])

    def get_result1(num1, num2, num3):
        top = max(num1, num2, num3)
        bot = min(num1, num2, num3)
        mid = num1 + num2 + num3 - top - bot
        return top + mid

    def get_discount(amount, rate):

    def get_discount_message(discount, rate):

    def print_docket(price, rate):
        print("Original price $" + str(price))
        disc = get_discount(price, rate)
        print(get_discount_message(disc, rate))
        print("Price $" + str(price - disc))
