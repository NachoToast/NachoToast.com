    def get_uniques_tuple(a_tuple):
        output = []
        for item in a_tuple:
            if item not in output:
                output.append(item)
        return tuple(output)

    def carry_out_transactions(balance, transactions_tuple):
        withdrawals = 0
        deposits = 0
        final_balance = balance + sum(transactions_tuple)
        for trans in transactions_tuple:
            if trans < 0:
                withdrawals -= trans
            else:
                deposits += trans
        return (final_balance, deposits, withdrawals)
