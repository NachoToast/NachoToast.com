    def get_number_of_uniques(a_sequence):
        uniques = []
        for letter in a_sequence:
            if letter not in uniques:
                uniques.append(letter)
        return len(uniques)

    def count_occurrences(a_sequence, value):
        count = 0
        for char in a_sequence:
            if char == value:
                count += 1
        return count

    def get_first_duplicate(a_sequence):
        uniques = []
        for char in a_sequence:
            if char in uniques:
                return char
            uniques.append(char)
        return -1

    def count_larger_element(a_sequence, letter):
        alphabet = "abcdefghijklmnopqrstuvwxyz"
        tested_value = alphabet.find(letter)
        count = 0
        for char in a_sequence:
            if alphabet.find(char) > tested_value:
                count += 1
        return count

    def count_doubles(a_sequence):
        check_dubs = []
        count = 0
        for char in a_sequence:
            if char in check_dubs:
                count += 1
                check_dubs.pop(check_dubs.index(char))
            else:
                check_dubs.append(char)
        return count
