    def write_to_file(filename, list1, list2):
        output_file = open(filename, "w")
        for i in range(len(list1)):
            output_file.write(str(list1[i]) + ": " + str(list2[i]) + "\n")
        output_file.close()

    def get_percent_vowels(filename):
        vowels = "aeiouAEIOU"
        vowel_count = 0
        letter_count = 0
        input_file = open(filename, "r")
        contents = input_file.read()
        for char in contents:
            if vowels.find(char) != -1:
                vowel_count += 1
            if char != " ":
                letter_count += 1
        input_file.close()
        return round( 100 * vowel_count / letter_count)

    def copy_file(filename):
        input_file = open(filename, "r")
        contents = input_file.read()
        first = contents[0]
        last = contents[-1]
        input_file.close()
        output_file = open("ldou933.txt", "w")
        output_file.write(contents)
        output_file.close()
        return first + last
