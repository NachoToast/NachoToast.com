    def to_the_power_of():
        num1 = int(input("Number 1: "))
        num2 = int(input("Number 2: "))
        return str(num1) + " to the power of " + str(num2) + " is " + str(num1 ** num2)

    def in_range(n, outside_mode):
        if outside_mode:
            if n < 1 or n > 10:
                return True
        else:
            if n >= 1 and n <= 10:
                return True
        return False

    def print_slices(word, index1, index2):
        word = word.upper()
        print(word[:min(index1, index2)] + " " + word[min(index1,index2):max(index2,index1)] + " " + word[max(index1,index2):])
