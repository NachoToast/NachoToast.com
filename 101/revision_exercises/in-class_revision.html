    def print_slices(word, index1, index2):
        word = word.upper()
        part1 = word[:min(index1, index2)]
        part2 = word[min(index1, index2):max(index1, index2)]
        part3 = word[max(index1, index2):]
        print(part1, part2, part3)

    def get_ordered_words(word1, word2):
        shorter = word1
        longer = word2
        if len(word2) < len(word1):
            shorter = word2
            longer = word1
        return longer + " " + shorter

    def fiddle_string(text):
        even = len(text) % 2 == 0
        if even:
            return text[len(text) // 2:] + text[:len(text)//2]
        return text[len(text) // 2 + 1:] + text[len(text) // 2] + text[:len(text) // 2]
