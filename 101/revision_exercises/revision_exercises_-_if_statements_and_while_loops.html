    def get_sum_of_divisors(number):
        divisor = 1
        div_sum = 0
        while divisor <= number // 2:

    def eliminate_repetition(param_str):

            if param_str.rfind(param_str[i]) != i:
                param_str = param_str[:param_str.rfind(param_str[i])] + param_str[param_str.rfind(param_str[i]) + 1:]
