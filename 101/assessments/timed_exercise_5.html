    def get_average_word_length(text):
        if len(text.split()) < 1:
            return 0.0
        sum_of_lengths = 0
        num_of_lengths = 0
        for word in text.split():
            sum_of_lengths += len(word)
            num_of_lengths += 1
        return round(sum_of_lengths / num_of_lengths, 2)

    def make_list_elements_multiples(int_list, number):
        for i in range(0, len(int_list)):
            if int_list[i] % number != 0:
                int_list[i] *= number
