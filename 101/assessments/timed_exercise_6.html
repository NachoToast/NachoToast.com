    def append_sums_to_front(list1, list2):
        sum1 = sum(list1)
        sum2 = sum(list2)
        list1.insert(0, sum2)
        list2.insert(0, sum1)

    def remove_negatives(numbers_list):
        i = 0
        while i < len(numbers_list):
            if numbers_list[i] < 0:
                numbers_list.pop(i)
            else:
                i += 1
