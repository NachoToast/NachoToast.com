    def extract_words(text):
        output_list = []
        while text.find(" ") != -1:
            text = text.strip()
            next_word = text[:text.find(" ")]
            if next_word[-3:].lower() == "ing":
                output_list += [next_word]
            text = text[text.find(" ") + 1:]
        if text[-3:].lower() == "ing":
            output_list += [text]
        return output_list

    def get_odd_multiples_of_three(numbers_list):
        output = []
        for number in numbers_list:
            if number % 2 != 0 and number % 3 == 0:
                output += [number]
        return output

    def get_value_from_list(a_list):
        print("List:", a_list)
        prompt = "Enter index of element you want to retrieve: "
        selected_index = int(input(prompt))
        while selected_index >= len(a_list) or selected_index < -len(a_list):
            print("The index you have entered is invalid.")
            selected_index = int(input(prompt))
        return a_list[selected_index]

    def make_all_odd(integer_list):
        if len(integer_list) < 1:
            return
        i = 0
        for num in integer_list:
            if num % 2 == 0:
                if num < 0:
                    integer_list[i] = num - 1
                else:
                    integer_list[i] = num + 1
            i += 1

    def convert_to_integers(value_list, boolean_list):
        i = 0
        for item in value_list:
            if boolean_list[i] == True:
                value_list[i] = int(item)
            else:
                value_list[i] = 0
            i += 1
