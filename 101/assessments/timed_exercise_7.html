    def analysis_numbers(numbers_list):
        negatives = 0
        zeros = 0
        positives = 0
        for num in numbers_list:
            if num < 0:
                negatives += 1
            elif num == 0:
                zeros += 1
            else:
                positives += 1
        return (negatives, zeros, positives)

    def get_word_pairs(filename):
        input_file = open(filename, "r")
        contents = input_file.read().split()
        input_file.close()
        i = 0
        output = []
        while i < len(contents) - 1:
            output.append((contents[i], contents[i+1]))
            i += 1
        return output
