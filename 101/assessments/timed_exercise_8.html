    def get_vowels_count(word):
        vowels = ['a', 'e', 'i' ,'o', 'u']
        vowels_count = 0
        for letter in word:
            if letter in vowels:
                vowels_count += 1
        return vowels_count

    def create_vowels_count_dictionary(sentence):
        a_dict = {0: [], 1: [], 2: [], 3: [], 4: [], 5: []}
        for word in sentence.split():
            vowel_count = get_vowels_count(word)
            if vowel_count in a_dict and word not in a_dict[vowel_count]:
                a_dict[vowel_count].append(word)
            elif vowel_count not in a_dict:
                a_dict[vowel_count] = [word]
        return a_dict
