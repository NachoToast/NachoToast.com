    def create_squares_dictionary(numbers):
        numbers.sort()
        a_dict = {}
        for num in numbers:
            a_dict[num] = num ** 2
        return a_dict

    def create_names_dictionary(names_tuple):
        names_tuple = list(names_tuple)
        names_tuple.sort()
        a_dict = {}
        for name in names_tuple:
            a_dict[name[0]] = name
        return a_dict

    def print_names_descending(names_dict):
        for key in sorted(names_dict, reverse=True):
            print(key, names_dict[key])

    def has_duplicated_values(dict1):
        vals = []
        for key in dict1:
            if dict1[key] in vals:
                return True
            vals.append(dict1[key])
        return False

    def get_unique_words(sentence):
        unique = []
        for word in sorted(sentence.split()):
            if word not in unique:
                unique += [word]
        return unique

    def count_word_occurrences(sentence):
        sentence = sentence.split()
        sentence.sort()
        a_dict = {}
        for word in sentence:
            if word not in a_dict:
                a_dict[word] = 1
            else:
                a_dict[word] += 1
        return a_dict

    def create_dog_breeds_dictionary(dog_breeds_list, dog_groups_list):
        a_dict = {}
        for i in range(len(dog_groups_list)):
            a_dict[dog_groups_list[i]] = dog_breeds_list[i].split(",")
        return a_dict
