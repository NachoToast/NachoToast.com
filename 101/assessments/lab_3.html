    def get_positive_integer():
        integer = input("Please enter a positive integer value: ")
        return int(integer)

    def calculate_cuboid_surface_area():
        length = get_positive_integer()
        width = get_positive_integer()
        height = get_positive_integer()
        print("A cuboid with a length of ", length, ", a width of ", width, " and a height of ", height, sep = "")
        surface_area = 2 * length * width
        surface_area += 2 * length * height
        surface_area += 2 * height * width
        print("has a surface area of", surface_area)

    def get_name():
        name = input("Please enter your name: ")
        return name

    def process_name(name):
        space = name.find(" ")
        first = name[0:1]
        last = name[space + 1: space + 4]
        return first.lower() + last.lower()

    def zero_extend_number(number, length):
        zeros_to_add = abs(length - len(str(number)))
        widened = "0" * zeros_to_add
        return widened + str(number)

    def generate_username(number, number_length):
        name = get_name()
        first_comp = process_name(name)
        second_comp = zero_extend_number(number, number_length)
        username = first_comp + second_comp
        print("Your username is", username)
