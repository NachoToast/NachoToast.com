    def get_string_distance(string1, string2):
        i = 0
        differences = 0
        len1 = len(string1)
        len2 = len(string2)
        while i < min(len1, len2):
            if string1[i] != string2[i]:
                differences += 1
            i += 1
        if len1 != len2:
            differences += max(len1, len2) - min(len1, len2)
        return differences

    def solve_equation(equation):
        operand1 = float(equation[:equation.find(" ")])
        operand2 = float(equation[equation.rfind(" ") + 1:])
        operator = equation[equation.find(" ") + 1:equation.rfind(" ")]
        
        if operator == "+":
            return round(operand1 + operand2, 3)
        if operator == "-":
            return round(operand1 - operand2, 3)
        if operator == "*":
            return round(operand1 * operand2, 3)
        if operator == "/":
            return round(operand1 / operand2, 3)
        if operator == "//":
            return round(operand1 // operand2, 3)
        if operator == "**":
            return round(operand1 ** operand2, 3)

    def make_palindrome(a_string):
        converted_string = ""
        i = 0
        while i < len(a_string):
            if a_string[i].isalpha() or a_string[i].isdigit():
                converted_string += a_string[i].lower()
            i +=1
        i = len(converted_string) - 1
        reversed_string = ""
        while i >= 0:
            reversed_string += converted_string[i]
            i -= 1
        if reversed_string == converted_string:
            return converted_string
        else:
            return converted_string + reversed_string
