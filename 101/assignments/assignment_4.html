    def get_lines_from_file(filename):
        input_file = open(filename, "r")
        contents = input_file.read().split()
        input_file.close()
        return contents

    def create_contacts(lines):
        a_dict = {}
        for string in lines:
            info = string.split(",")
            a_dict[info[0]] = [info[1], info[2]]
        return a_dict

    def update_contact(contacts_dictionary, name, new_phone, new_email):
        if name not in contacts_dictionary:
            return print(name + " is NOT found.")
        print(name + "'s contact has been updated.")
        contacts_dictionary[name] = [new_phone, new_email]
