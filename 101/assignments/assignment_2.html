    def roll_three_dice():
        dice1 = random.randrange(1, 7)
        dice2 = random.randrange(1, 7)
        dice3 = random.randrange(1, 7)
        return str(dice1) + str(dice2) + str(dice3)

    def is_456(dice_str):
        if dice_str.find("4") != -1 and dice_str.find("5") != -1 and dice_str.find("6") != -1:
            return True
        else:
            return False

    def is_123(dice_str):
        if dice_str.find("1") != -1 and dice_str.find("2") != -1 and dice_str.find("3") != -1:
            return True
        else:
            return False

    def is_trip(dice_str):
        if dice_str[0] == dice_str[1] and dice_str[1] == dice_str[2]:
            return True
        else:
            return False

    def is_point(dice_str):
        if dice_str[0] == dice_str[1] == dice_str[2]:
            return False
        elif dice_str[0] == dice_str[1] or dice_str[1] == dice_str[2] or dice_str[0] == dice_str[2]:
            return True
        else:
            return False

    def is_valid_roll(dice_str):
        if is_123(dice_str) or is_456(dice_str) or is_point(dice_str) or is_trip(dice_str):
            return True
        else:
            return False

    def get_valid_roll():
        roll = roll_three_dice()
        while is_valid_roll(roll) == False:
            roll = roll_three_dice()
        return roll

    def type_of_roll(dice_str, player):
        if is_123(dice_str):
            return player + " has rolled a 123"
        elif is_456(dice_str):
            return player + " has rolled a 456"
        elif is_point(dice_str):
            return player + " has rolled a POINT"
        elif is_trip(dice_str):
            return player + " has rolled a TRIP"

    def get_point_score(dice_str):
        score = 10
        if dice_str[0] == dice_str[1]:
            score += int(dice_str[2])
        elif dice_str[0] == dice_str[2]:
            score += int(dice_str[1])
        elif dice_str[1] == dice_str[2]:
            score += int(dice_str[0])
        return score

    def get_trip_score(dice_str):
        score = 20 + int(dice_str[0])
        return score

    def get_score(dice_str):
        if is_trip(dice_str):
            return get_trip_score(dice_str)
        if is_point(dice_str):
            return get_point_score(dice_str)
        if is_456(dice_str):
            return 30
        return 0

    def print_roll(player_name, player_roll, player_score):
        print(player_name, "has rolled:", player_roll)
        print("(", type_of_roll(player_roll, player_name), " for a score of ", player_score, ")", sep = "")

    def print_player_stats(player_wins, computer_wins, draws):
        if (player_wins + computer_wins + draws) > 0:
            win_percentage = 100 * player_wins / (player_wins + computer_wins + draws)
        else:
            win_percentage = 0.0
        print("Player wins:", player_wins)
        print("Win percentage: ", round(win_percentage, 1), "%", sep = "")
